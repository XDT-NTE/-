	1) 将 id 设置为主键、自增值约束
		alter table author change id id int primary key auto_increment;

		alter table author change note note text default('当前没啥好说的。');

		alter table author change tele telephone varchar(20);

三、索引
1. 在创建表的时候创建索引
	create table 表名 (
		属性名1 数据类型 [完整性约束条件],
		属性名2 数据类型 [完整性约束条件],
		...
		属性名 数据类型 [完整性约束条件],
		[unique|fulltext|spatial] index|key [别名](属性名1[(长度)][ASC|DESC])
	)engine=InnoDB character set utf8 collate utf8_bin;
	
	1) 创建普通索引(不需要加任何unique,fulltext,spatial参数)
		***例：创建一个表index1，在表中的id字段上建立索引
		create table index1 (
			id int,
			name varchar(20),
			sex boolean,
			index(id)
		)engine=InnoDB character set utf8 collate utf8_bin;
	
		show create table index1;
		explain select * from index1 where id=1 \G;
	
	{
		https://blog.csdn.net/lvhaizhen/article/details/90763799
		
		mysql 用法 Explain
		explain（执行计划），使用explain关键字可以模拟优化器执行sql查询语句，从而知道MySQL是如何处理sql语句。explain主要用于分析查询语句或表结构的性能瓶颈
	。
		A.explain的作用
		通过explain+sql语句可以知道如下内容：

		①表的读取顺序。（对应id）

		②数据读取操作的操作类型。（对应select_type）

		③哪些索引可以使用。（对应possible_keys）

		④哪些索引被实际使用。（对应key）

		⑤表直接的引用。（对应ref）

		⑥每张表有多少行被优化器查询。（对应rows）

		B.explain包含的信息
		explain使用：explain+sql语句，通过执行explain可以获得sql语句执行的相关信息。
	}
	
	2) 创建唯一性索引
		创建唯一性索引时，需要使用 UNIQUE 参数进行约束。
		create table index2 (
			id int,
			name varchar(20),
			unique index index2_id(id ASC)
		                创建唯一索引  表的id字段 （ASC:升序/DESC:降序；两个是可选参数）
		)engine=InnoDB character set utf8 collate utf8_bin;
	
	
	3) 创建全文索引
		全文索引只能创建在 （“char" or  "varchar" or "text" 类型的字段上，而且，现在只有 ((((MyISAM 存储引擎))))支持全文索引。
		create table index3 (
			id int,
			info varchar(20),
			fulltext index index3_info(info)
			全文索引  index  索引名（字段名）
		)engine=MyISAM character set utf8 collate utf8_bin;
		在创建全文索引时一定要注意表的存储引擎的类型，对于经常需要索引的字符串、文字数据等信息，可以考虑存储到 MyISAM 存储引擎的表中。
	
	4) 创建单列索引
		单列索引是在表的单个字段上创建索引。
		create table index4 (
			id int,
			subject varchar(30),
			index index4_su(subject(10))
			单列索引  索引名 （字段名）      
		)engine=InnoDB character set utf8 collate utf8_bin;
		index4_st 索引的长度只有10，这样做的目的还是为了提高查询速度，对于字符型的数据，可以不用查询全部信息，而只查询其前面的若干字符信息。
	
	5) 创建多列索引
		创建多列索引是在表的多个字段上创建一个索引，多列查询中，只有查询条件中使用了这些字段中第一个字段时，索引才会被使用。
		create table index5 (
			id int,
			name varchar(20),
			sex char(4),
			index index5_ns(name,sex)
			index  索引名 （索引的字段）
		)engine=InnoDB character set utf8 collate utf8_bin;
	
	
	6) 创建空间索引
		创建空间索引时必须使用 spatial 参数来设置，创建空间索引时，表的存储引擎必须是 MyISAM 类型，而且，(((索引字段必须有非空约束)))。
		create table index6 (
			id int,
			space geometry not null,
			spatial index index6_sp(space)
			空间索引  index  索引名(space字段）
		)engine=MyISAM character set utf8 collate utf8_bin;
	
	
2. 在已经存在的表上创建索引
	在已经存在的表中，可以直接为表上的一个或几个字段创建索引，基本形式如下：
	create [unique|fulltext|spatial] index 索引名 on 表名(属性名 [(长度)][ASC|DESC]);
	其中：
		UNIQUE 是可选参数，表示索引为唯一性索引；
		FULLTEXT 是可选参数，表示索引为全文索引；
		SPATIAL 也是可选参数，表示索引为空间索引；
		“index” 参数用来指定字段为索引；
		“索引名”参数是给创建的索引取的新名称；
		“表名”参数是指需要创建索引的表的名称，该表必须是已经存在的，如果不存在，则需要先创建；
		“属性名”参数指定索引对应的字段的名称，该字段必须为前面定义好的字段；
		“长度”是可选参数，其指索引的长度，必须是字符串类型才可以使用；
		ASC|DESC也是可选参数，表示“升序｜降序”排序。
	
	1) 创建普通索引
		use 21dsj2;
		create index index7_id on zuozhe(z_id);
		creste   索引  索引名    在     表名（字段名）
	2) 创建唯一性索引
		create unique index index8_id on cjb1(xuhao);
		create  唯一索引 index 索引名   on   表 （字段名）
	3) 创建全文索引
		create fulltext index index9_adr on zuozhe(addr);
		create 全文索引 index 索引名 on  表名（字段名）
	4) 创建单列索引
		create index index10_addr on zuozhe(addr(4));
		create 单行索引 索引名     on 表名 （字段名）
	5) 创建多列索引
		create index index11_na on zuozhe(name,addr);
		 create 多列索引 索引名  on 表名（所需索引的字段名）
	6) 创建空间索引
		create spatial index index12_line on index12(line);
		create 空间索引 index  索引名     on   表名（字段名）
		注意：SPATIAL用来设置索引为空间索引，表index12的存储引擎必须是 MyISAM 类型，而且line字段必须是空间数据类型，而且是非空的。
		

3. 用 alter table 语句来创建索引
	在已经存在的表上，可以通过 alter table 语句直接为表上的一个或几个字段创建索引，基本形式如：
	alter table 表名 add [unique|fulltext|spatial] index 索引名 (属性名 [(长度)][ASC|DESC]);
	
	1) 创建普通索引
		alter table example0 add index idx13_name(name(20));
		alter table 表名    add  index  索引名 （索引字段名）
	2) 创建唯一性索引
		alter table example0 add unique index idx14_id(course_id);
		alter table 表名   add  唯一索引  index  索引名（字段名）
	3) 创建全文索引
		alter table example0 add fulltext index idx15_info(info);
		alter table 表名    add  全文索引  index   索引名  （字段名）
	4) 创建单列索引
		alter table example0 add index idx16_addr(address(4));
		alter table 表名   add    单列索引  索引名（字段名）
	5) 创建多列索引
		alter table example0 add index idx17_na(name,address);
		alter table 表名   add  多列索引  索引名（所需索引的字段名）
	6) 创建空间索引
		alter table index17 add spatial index idx18_line(line);
		alter table 表名   add   空间索引  index  索引名（字段名）
4. 删除索引
	删除索引是指将表中已经存在的索引删除掉，一些不再使用的索引会降低表的更新速度，影响数据库的性能，对于这样的索引，应该将其删除。
	对于已经存在的索引，可以通过 DROP 语句来删除索引，基本形式如下：
	drop index 索引名 on 表名;
	删除索引之前，先通过 show create table 表名; 语句来查看存在的索引的名称。
